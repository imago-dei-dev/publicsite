Overview

This project focuses on building Python scripts using boto3 to automate AWS enumeration, auditing, and evidence collection tasks. The goal is to create reusable tools that can eventually feed into automated compliance evidence gathering for RMF frameworks.

Completed Scripts

1. S3 Bucket Lister with Encryption Check

import boto3

def list_buckets_with_encryption():
    s3 = boto3.client('s3')
    buckets = s3.list_buckets()['Buckets']
    
    for bucket in buckets:
        name = bucket['Name']
        try:
            encryption = s3.get_bucket_encryption(Bucket=name)
            print(f"{name}: Encryption enabled")
        except:
            print(f"{name}: NO ENCRYPTION")
            
if __name__ == "__main__":
    list_buckets_with_encryption()

2. EC2 Instance Describer with Security Group Output

import boto3

def describe_ec2_instances():
    ec2 = boto3.client('ec2')
    instances = ec2.describe_instances()
    
    for reservation in instances['Reservations']:
        for instance in reservation['Instances']:
            print(f"Instance ID: {instance['InstanceId']}")
            print(f"State: {instance['State']['Name']}")
            print("Security Groups:")
            for sg in instance['SecurityGroups']:
                print(f"  - {sg['GroupName']} ({sg['GroupId']})")
            print("-" * 40)
